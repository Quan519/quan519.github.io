由于数据范围是 $0 \le n \le 2 \times 10^5$，要求的时间复杂度为 $O(n \log n)$，空间复杂度为 $O(n)$，最适合采用**贪心算法（Greedy Algorithm）配合排序**来解决。对于进阶要求，由于区间内的数值范围 $0 \le val \le 2 \times 10^5$ 较小，可以采用**差分数组 / 计数排序思想**来实现 $O(val)$ 的时空复杂度。

## 思路

### 1. 定义状态

为了方便合并，首先需要将所有区间按照**起点升序**进行排序。如果起点相同，则按终点升序排列。

排序后，我们用一个结果集 `res` 来存合并后的区间。

### 2. 状态转移方程（核心合并逻辑）

从左到右遍历排序后的区间列表。对于当前遍历到的区间 `current`：

- 如果 `res` 为空，或者 `current.start` 大于 `res` 中最后一个区间的 `end`（即 `res[last].end < current.start`），说明这两个区间**没有重叠**，直接将 `current` 放入 `res`。

- 如果 `current.start` 小于或等于 `res` 中最后一个区间的 `end`，说明这两个区间**发生重叠**。我们需要合并它们，即将 `res` 中最后一个区间的 `end` 更新为它们两个终点的最大值：

    $$\text{res[last].end} = \max(\text{res[last].end}, \text{current.end})$$

### 3. 初始状态与边界

如果 $n = 0$，没有区间需要合并，直接返回空数组。

### 4. 最终答案

遍历完所有区间后，`res` 中存储的就是合并后且按起点升序排列的所有区间。

## 进阶思路：$O(val)$ 差分/前缀和解法

当 $val$ 的范围不大时，我们可以利用一个大小为 $2 \times 10^5 + 2$ 的数组 `diff`。

- 每一个区间的起点 `start`，令 `diff[start * 2]++`

- 每一个区间的终点 `end`，令 `diff[end * 2 + 1]--`

    _(注：乘2是为了处理类似 `[1,1]` 这样单点区间以及相邻区间的边界重叠问题。)_

    通过从左到右累加这个数组（求前缀和），当累加值从 `0` 变为大于 `0` 时即为合并区间的起点，当累加值从大于 `0` 变回 `0` 时即为合并区间的终点。

## Go 语言源码实现

```go
package main

import (
	"fmt"
	"sort"
)

// Interval 结构体定义
type Interval struct {
	Start int
	End   int
}

// 基础解法：O(n log n) 排序法
func merge(intervals []Interval) []Interval {
	n := len(intervals)
	if n == 0 {
		return []Interval{}
	}

	// 1. 按照起点升序排序
	sort.Slice(intervals, func(i, j int) bool {
		if intervals[i].Start == intervals[j].Start {
			return intervals[i].End < intervals[j].End
		}
		return intervals[i].Start < intervals[j].Start
	})

	// 2. 遍历并合并
	res := make([]Interval, 0)
	res = append(res, intervals[0])

	for i := 1; i < n; i++ {
		lastIdx := len(res) - 1
		// 如果当前区间的起点小于等于上一个区间的终点，说明有重叠，进行合并
		if intervals[i].Start <= res[lastIdx].End {
			if intervals[i].End > res[lastIdx].End {
				res[lastIdx].End = intervals[i].End
			}
		} else {
			// 没有重叠，直接添加新区间
			res = append(res, intervals[i])
		}
	}

	return res
}

func main() {
	// 测试用例
	intervals := []Interval{
		{1, 4},
		{2, 5},
		{7, 9},
		{3, 6},
	}

	fmt.Println("原区间:", intervals)
	res := merge(intervals)
	fmt.Println("合并后:", res) // 输出: [{1 6} {7 9}]
}
```

## C++ 语言源码实现

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// Interval 结构体定义
struct Interval {
    int start;
    int end;
};

class Solution {
public:
    // 基础解法：O(n log n) 排序法
    vector<Interval> merge(vector<Interval>& intervals) {
        if (intervals.empty()) return {};

        // 1. 按照起点升序排序
        sort(intervals.begin(), intervals.end(), [](const Interval& a, const Interval& b) {
            if (a.start == b.start) return a.end < b.end;
            return a.start < b.start;
        });

        vector<Interval> res;
        res.push_back(intervals[0]);

        // 2. 遍历并合并
        for (size_t i = 1; i < intervals.size(); ++i) {
            // 如果当前区间的起点小于等于上一个区间的终点，说明有重叠
            if (intervals[i].start <= res.back().end) {
                res.back().end = max(res.back().end, intervals[i].end);
            } else {
                // 没有重叠，直接放入结果集
                res.push_back(intervals[i]);
            }
        }

        return res;
    }

    // 进阶解法：O(val) 计数/前缀和法
    vector<Interval> mergeAdvanced(vector<Interval>& intervals) {
        if (intervals.empty()) return {};

        const int MAX_VAL = 200005;
        // 使用两倍大小处理边界闭区间问题
        vector<int> diff(MAX_VAL * 2, 0);

        for (const auto& kv : intervals) {
            diff[kv.start * 2]++;
            diff[kv.end * 2 + 1]--;
        }

        vector<Interval> res;
        int count = 0;
        int start = -1;

        for (int i = 0; i < MAX_VAL * 2; ++i) {
            count += diff[i];
            if (count > 0 && start == -1) {
                start = i / 2; // 记录起点
            }
            // 只有当 count 回落到 0，且下一位不会马上把它拉起来时，才是一个完整区间的终点
            if (count == 0 && start != -1) {
                res.push_back({start, i / 2});
                start = -1;
            }
        }
        return res;
    }
};

int main() {
    Solution sol;
    vector<Interval> intervals = {{1, 4}, {2, 5}, {7, 9}, {3, 6}};

    vector<Interval> res = sol.merge(intervals);

    cout << "合并后区间: " << endl;
    for (const auto& kv : res) {
        cout << "[" << kv.start << ", " << kv.end << "] ";
    }
    cout << endl; // 输出: [1, 6] [7, 9]

    return 0;
}
```

## 复杂度分析

### 基础解法

- **时间复杂度**：$O(n \log n)$。主要耗时在对长度为 $n$ 的区间数组进行排序。之后的单层 `for` 循环遍历的时间复杂度为 $O(n)$。总体复杂度由排序决定，为 $O(n \log n)$。

- **空间复杂度**：$O(n)$。我们创建了一个 `res` 数组来存储合并后的区间，最坏情况下（没有任意两个区间重叠），空间开销为 $O(n)$。

### 进阶解法

- **时间复杂度**：$O(val)$。投射区间到差分数组需要 $O(n)$，遍历大小为 $2 \times val$ 的标准数组需要 $O(val)$。由于 $n$ 与 $val$ 处于同数量级，整体时间复杂度收敛在 $O(val)$。

- **空间复杂度**：$O(val)$。使用了一个固定大小为 $2 \times val$ 的辅助数组。
